# a) Experiments on Solid images
# they are with
# OPT=2

for I in {64,128,256}
do
  echo "============== "$I" ================="
  echo "BLOCK "
  echo FileSolid/"block"$I"_cubes.txt"
  python3 -m cProfile main_for_tests.py 3 2 FileSolid/"block"$I"_cubes.txt" 10 > out_block$I
  echo "BUMPYTORUS"
  python3 -m cProfile main_for_tests.py 3 2 FileSolid/"bumpytorus"$I"_cubes.txt" 10 > out_bumpytorus$I
  echo "CAMEL"
  python3 -m cProfile main_for_tests.py 3 2 FileSolid/"camel"$I"_cubes.txt" 10 > out_camel$I
  echo "CUP"
  python3 -m cProfile main_for_tests.py 3 2 FileSolid/"cup"$I"_cubes.txt" 10 > out_cup$I
  echo "PEGASUS"
  python3 -m cProfile main_for_tests.py 3 2 FileSolid/"pegasus"$I"_cubes.txt" 10 > out_pegasus$I
  echo "TEAPOT"
  python3 -m cProfile main_for_tests.py 3 2 FileSolid/"teapot"$I"_cubes.txt" 10 > out_teapot$I
done
